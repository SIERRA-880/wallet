USER_PASSCLI(PRIMARY KEY (natID))

CLIENT(PRIMARY KEY (userID),
		FOREIGN KEY (userID) REFERENCES WALLETS,
		FOREIGN KEY (userID) REFERENCES CLIEN_VS_INST,
		FOREIGN KEY (natID) REFERENCES USER_PASSCLI,
		UNIQUE (natID))

USER_PASSINST(PRIMARY KEY (bic))

CLIENT_VS_INST(PRIMARY KEY(bic),
				FOREIGN KEY(bic) REFERENCES USER_PASSINST,
				UNIQUE (userID))

WALLETS(PRIMARY KEY (walletID),
		FOREIGN KEY (userID) REFERENCES CLIENT)

ACCOUNTS(PRIMARY KEY (walletID),
		FOREIGN KEY (walletID) REFERENCES WALLETS,
		FOREIGN KEY (userID) REFERENCES CLIENT_VS_INST,
		UNIQUE(iban))

CO_OWNER(PRIMARY KEY(ibanOwner, walletID-coOwner),
		FOREIGN KEY(ibanOwner) REFERENCES ACCOUNTS,
		FOREIGN KEY(walletID_coOwner) REFERENCES WALLETS)

CASH_BALANCES(PRIMARY KEY(balanceID),
		      FOREIGN KEY(iban) references ACCOUNTS,
			  UNIQUE(currency))

CASH_MVT(PRIMARY KEY(iban),
		FOREIGN KEY(banlanceID) REFERENCES CASH_BALANCES,
		UNIQUE(balanceID))

TRANSACTIONS(PRIMARY KEY(streamID, trxID),
		     FOREIGN KEY(iban) REFERENCES ACCOUNTS,
			 FOREIGN KEY(iban) REFERENCES CASH_MVT)
